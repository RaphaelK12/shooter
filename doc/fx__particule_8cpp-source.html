00027 // Mise à jour de la position de l'entité, à l'aide des équations de mouvement uniformément accéléré
00028 // x(t) = 1/2*a*(t-to)² + v(t)*(t-to) + x0
00029 // v(t) = a(t)*(t-to) + v0
00030 // a(t) = a0 (constante)
00031 CVecteur CParticule::GetPosition () const
00032 {
00033         // Détermine une durée c'est à dire la différence entre le temps initiale et le temps finale (t - to)
00034         float fTemps = m_Temps.GetTime ();
00035         // Détermine la vitesse : v(t) = a(t)*(t-to) + v0
00036         CVecteur m_VitesseF = m_Acceleration * fTemps + m_Vitesse;
00037         // Détermine la position : x(t) = 1/2*a*(t-to)² + v(t)*(t-to) + x0
00038         return m_Acceleration * 0.5 * SQR(fTemps) + m_VitesseF * fTemps + m_Position;
00039 }
